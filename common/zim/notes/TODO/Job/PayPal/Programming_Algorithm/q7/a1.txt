import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;

public class dictionary {

    HashMap<String, List<String>> firstNameMap = new HashMap<String, List<String>>();
    HashMap<String, List<String>> lastNameMap = new HashMap<String, List<String>>();

    public boolean add(String name) {

      if(name == null || name.isEmpty()) {
        return false;
      }

      String[] stringArr = name.split("\\s+");

      String fName = stringArr[0];
      String lName = stringArr[1];

      if(firstNameMap.containsKey(fName)) {
        List<String> temp = firstNameMap.get(fName);
        temp.add(name);
      } else {
        List<String> temp = new ArrayList<String>();
        temp.add(name);
        firstNameMap.put(fName, temp);
      }

      if(lastNameMap.containsKey(lName)) {
        List<String> temp = lastNameMap.get(lName);
        temp.add(name);
      } else {
        List<String> temp = new ArrayList<String>();
        temp.add(name);
        lastNameMap.put(lName, temp);
      }

      return true;

    }

    public List<String> searchByLastName(String lname) {
      if(lname == null) return null;
      return lastNameMap.get(lname);
    }

    public List<String> searchByFirstName(String fname) {
      if(fname == null) return null;
      return firstNameMap.get(fname);
    }
}
